#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define MAX_REMINDERS 10
#define MAX_STACK 50
struct Medicine {
    char name[50];
    int dose;
};

struct AVLNode {
    struct Medicine med;
    struct AVLNode* left;
    struct AVLNode* right;
    int height;
};

int max(int a, int b) { return (a > b) ? a : b; }

struct AVLNode* newAVLNode(char* name, int dose) {
    struct AVLNode* node = (struct AVLNode*)malloc(sizeof(struct AVLNode));
    strcpy(node->med.name, name);
    node->med.dose = dose;
    node->left = node->right = NULL;
    node->height = 1;
    return node;
}

struct AVLNode* rightRotate(struct AVLNode* y) {
    struct AVLNode* x = y->left;
    struct AVLNode* T2 = x->right;
    x->right = y;
    y->left = T2;
    y->height = max(height(y->left), height(y->right)) + 1;
    x->height = max(height(x->left), height(x->right)) + 1;
    return x;
}
struct AVLNode* leftRotate(struct AVLNode* x) {
    struct AVLNode* y = x->right;
    struct AVLNode* T2 = y->left;
    y->left = x;
    x->right = T2;
    x->height = max(height(x->left), height(x->right)) + 1;
    y->height = max(height(y->left), height(y->right)) + 1;
    return y;
}
int getBalance(struct AVLNode* node) {
    return (node == NULL) ? 0 : height(node->left) - height(node->right);
}
struct AVLNode* insertMedicine(struct AVLNode* node, char* name, int dose) {
    if (node == NULL) return newAVLNode(name, dose);
    if (strcmp(name, node->med.name) < 0)
        node->left = insertMedicine(node->left, name, dose);
    else if (strcmp(name, node->med.name) > 0)
        node->right = insertMedicine(node->right, name, dose);
    else return node;

    node->height = 1 + max(height(node->left), height(node->right));
    int balance = getBalance(node);

    if (balance > 1 && strcmp(name, node->left->med.name) < 0) return rightRotate(node);
    if (balance < -1 && strcmp(name, node->right->med.name) > 0) return leftRotate(node);
    if (balance > 1 && strcmp(name, node->left->med.name) > 0) {
        node->left = leftRotate(node->left);
        return rightRotate(node);
    }
    if (balance < -1 && strcmp(name, node->right->med.name) < 0) {
        node->right = rightRotate(node->right);
        return leftRotate(node);
    }
    return node;
}
struct Reminder {
    char medicineName[50];
    char time[10];
};

struct ReminderQueue {
    struct Reminder reminders[MAX_REMINDERS];
    int front, rear;
};

void initQueue(struct ReminderQueue* q) {
    q->front = q->rear = -1;
}

int isQueueEmpty(struct ReminderQueue* q) {
    return q->front == -1;
}

int isQueueFull(struct ReminderQueue* q) {
    return (q->rear + 1) % MAX_REMINDERS == q->front;
}

void enqueue(struct ReminderQueue* q, char* name, char* time) {
    if (isQueueFull(q)) {
        printf("Reminder queue full!\n");
        return;
    }
    if (isQueueEmpty(q)) q->front = 0;
    q->rear = (q->rear + 1) % MAX_REMINDERS;
    strcpy(q->reminders[q->rear].medicineName, name);
    strcpy(q->reminders[q->rear].time, time);
}

struct Reminder* peek(struct ReminderQueue* q) {
    if (isQueueEmpty(q)) return NULL;
    return &q->reminders[q->front];
}

void dequeue(struct ReminderQueue* q) {
    if (isQueueEmpty(q)) return;
    if (q->front == q->rear) q->front = q->rear = -1;
    else q->front = (q->front + 1) % MAX_REMINDERS;
}
struct Stack {
    char actions[MAX_STACK][100];
    struct Reminder undoReminders[MAX_STACK];
    int top;
};

void initStack(struct Stack* s) {
    s->top = -1;
}

void push(struct Stack* s, char* action, struct Reminder r) {
    if (s->top == MAX_STACK - 1) return;
    strcpy(s->actions[++(s->top)], action);
    s->undoReminders[s->top] = r;
}

void popUndo(struct Stack* s, struct ReminderQueue* q) {
    if (s->top == -1) {
        printf("No action to undo.\n");
        return;
    }
    enqueue(q, s->undoReminders[s->top].medicineName, s->undoReminders[s->top].time);
    printf("Undo: %s\n", s->actions[s->top]);
    s->top--;
}

void showHistory(struct Stack* s, const char* label) {
    if (s->top == -1) {
        printf(" No %s history available.\n", label);
        return;
    }
    printf("\n %s History:\n", label);
    for (int i = s->top; i >= 0; i--) {
        printf("   - %s\n", s->actions[i]);
    }
}

    struct AVLNode* medRoot = NULL;

    medRoot = insertMedicine(medRoot, "Paracetamol", 500);

    enqueue(&rq, "Paracetamol", "08:00 AM");
    enqueue(&rq, "Ibuprofen", "01:00 PM");
    enqueue(&rq, "Aspirin", "09:00 PM");

    char times[3][10] = {"08:00 AM", "01:00 PM", "09:00 PM"};

    while (!isQueueEmpty(&rq)) {
        struct Reminder* current = peek(&rq);
        printf("\n Next scheduled medicine: %s at %s\n", current->medicineName, current->time);

        printf(" Choose medicine you took:\n1. Paracetamol\n2. Ibuprofen\n3. Aspirin\n4. Undo Last Action\nYour choice (1-4): ");
        int medChoice;
        scanf("%d", &medChoice);
        if (medChoice == 4) {
            popUndo(&todayHistory, &rq);
            continue;
        }
        char userMed[50];
        if (medChoice == 1) strcpy(userMed, "Paracetamol");
        else if (medChoice == 2) strcpy(userMed, "Ibuprofen");
        else if (medChoice == 3) strcpy(userMed, "Aspirin");
        else {
            printf(" Invalid choice!\n");
            continue;
        }
        printf("Select time you took it:\n1. 08:00 AM\n2. 01:00 PM\n3. 09:00 PM\nYour choice (1-3): ");
        scanf("%d", &timeChoice);
        if (timeChoice < 1 || timeChoice > 3) {
            printf(" Invalid time choice!\n");
            continue;
        }
        char userTime[10];
        strcpy(userTime, times[timeChoice - 1]);

        char log[100];
        if (strcasecmp(userMed, current->medicineName) == 0 &&
            strcasecmp(userTime, current->time) == 0) {
            printf(" Great! You took %s at the correct time!\n", userMed);
            sprintf(log, "Took %s at %s", userMed, userTime);
            push(&todayHistory, log, *current);
            dequeue(&rq);
        } else {
            printf(" Warning: Wrong medicine or wrong time!\n");
            sprintf(log, "Missed %s at %s", current->medicineName, current->time);
            push(&todayHistory, log, *current);
            dequeue(&rq);
        }
    }
    int choice;
    printf("\n Do you want to view:\n1. Today's History\nEnter your choice: ");
    scanf("%d", &choice);
    if (choice == 1)
        showHistory(&todayHistory, "Today's");
    else
        printf(" Invalid choice!\n");
    return 0;
}
